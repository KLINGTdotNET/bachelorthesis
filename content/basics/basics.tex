\section{XML}


\begin{lstlisting}[caption="EBNF eines XML-Elementes"]
Element ::= "<" , Tagname , {Attribute , Whitespace} , ">" ,  
                Element , 
            "</" , Tagname , ">" 
            | "<" , Tagname , "/>";
Attribute ::= Key , "=" , Value | "";
Tagname ::= {UTF8-character};
Key ::= {UTF8-character};
Value ::= {UTF8-character};
\end{lstlisting}

\end{lstlisting}
