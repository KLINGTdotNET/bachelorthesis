\newcommand{\blankpage}[1]{
    \newpage
    \thispagestyle{empty}
    \begin{center}
    {
        \small
        \color{gray}
        #1
    }
    \end{center}
    \mbox{}
    \newpage
}

\newcommand{\chapterQuote}[4]{
    \epigraph{
        \enquote{#1}
    }
    {
        {
            \scriptsize
            \textit{
            {#4}\\
            \vspace{-3pt}
            {\textbf{#2}} ({#3})
            }
        }
    }        
}


\DeclareUrlCommand\noUnderscoreEscaping{\urlstyle{rm}} % needs url package

\newcommand{\printhref}[2]{
    \href{#1}{#2}
    \footnote{
        \noUnderscoreEscaping{#1}
    }
}

    \hspace{4pt}\hrulefill
}
