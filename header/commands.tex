%------------------------------------------------------------------------------
% Absichtlich leer gelassene Seiten werden auch "Vakatseiten" genannt. http://de.wikipedia.org/wiki/Vakatseite
%------------------------------------------------------------------------------
\newcommand{\blankpage}[1]{
    \newpage
    \thispagestyle{empty}
    \begin{center}
    {
        \small
        \color{gray}
        #1
    }
    \end{center}
    \mbox{}
    \newpage
}

%------------------------------------------------------------------------------
% quotation style for epigraph
%------------------------------------------------------------------------------
\newcommand{\chapterQuote}[4]{
    \epigraph{
        \enquote{#1}
    }
    {
        {
            \scriptsize
            \textit{
            {#4}\\
            \vspace{-3pt}
            {\textbf{#2}} ({#3})
            }
        }
    }        
}

%------------------------------------------------------------------------------
% decorative hrule under chapter quote (epigraph)
%------------------------------------------------------------------------------
\makeatletter
\renewcommand{\@epirule}{\hrulefill\hspace{4pt}
        \raisebox{-2pt}[10pt][10pt]{\adfflourishleftdouble}
    \hspace{4pt}\hrulefill}
\makeatother

%------------------------------------------------------------------------------
% decorative hrule used for pagestyle definition
%------------------------------------------------------------------------------
\newcommand{\decorativeLine}[2]{
    \hrulefill\hspace{4pt}
        \raisebox{-#2}[10pt][10pt]{#1}
    \hspace{4pt}\hrulefill
}

%------------------------------------------------------------------------------
% hack to escape underscores
%------------------------------------------------------------------------------
\DeclareUrlCommand\noUnderscoreEscaping{\urlstyle{}} % needs url package

%------------------------------------------------------------------------------
% href that prints the url into the footline
%------------------------------------------------------------------------------
\newcommand{\printhref}[2]{
    \href{#1}{#2}
    \footnote{\texttt{\noUnderscoreEscaping{#1}}}
}

%------------------------------------------------------------------------------
% quotation style
%------------------------------------------------------------------------------
\newcommand{\thesisquote}[1]{
    {
        \singlespacing
        \begin{quote}
            %\rule{\linewidth}{0.5pt}
            \textsl{\enquote{#1}}
            %\\
            %\rule{\linewidth}{0.5pt}
        \end{quote}
    }
}

%------------------------------------------------------------------------------
% definition style
%------------------------------------------------------------------------------
\newcommand{\thesisDefinition}[2]{
    \begin{leftbar}
        \begin{myDefinition}[#1]
            {
            \small
            #2
            }
        \end{myDefinition}
    \end{leftbar}
}
